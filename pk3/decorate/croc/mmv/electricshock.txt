actor ElectricShockWep : BaseMM8BDMWep 10106
{
    //$Category MM8BDM-Weapons/Rapid
    //$Title Electric Shock
    //$Sprite ESPAA0
    Weapon.AmmoUse 1
    Weapon.AmmoGive 56
    Weapon.SlotNumber 3
    Inventory.Pickupmessage "$PU_ELECTRICSHOCK"
    Obituary "$OB_ELECTRICSHOCK"
    Tag "$TAG_ELECTRICSHOCK"
    weapon.ammotype "ElectricShockAmmo"
    inventory.icon "ESPAWI"
    States
    {
    SpawnLoop:
        ESPA A -1
        loop
    Ready:
        ESPA D 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_ELECTRICSHOCK)
        ESPA D 1 A_WeaponReady
        goto Ready+1
    Deselect:
        ESPA D 0
        goto DeselectSwap
    Select:
        ESPA D 0
        goto SelectSwap
    Altfire:
    Fire:
        ESPA D 0 A_JumpIfNoAmmo("NoAmmo")
        ESPA EF 2
        TNT1 A 2
        ESPA GH 1
    AltHold:
    Hold:
        ESPA D 0 A_PlaySoundEx("weapons/mmv/electricshockfire", "Weapon")
        ESPA D 0 A_FireCustomMissile("ElectricShockBurstStart",0,0,0,0)
        ESPA I 3 A_FireCustomMissile("ElectricShockDamager",0,1,0,0)
        ESPA D 0 A_JumpIfNoAmmo("FireEnd")
        ESPA I 0 A_Refire
    FireEnd:
        ESPA JK 7
        ESPA D 0 A_ClearRefire
        goto Ready+1
    NoAmmo:
        ESPA D 1 ACS_NamedExecuteAlways("core_noammo",0)
        Goto Ready+1
    }
}
