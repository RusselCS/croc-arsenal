actor WaterWaveWep : BaseMM8BDMWep 10063
{
//$Category MM8BDM-Weapons/Rapid
//$Title Water Wave
//$Sprite WEA2M0
Weapon.AmmoUse 2
Weapon.AmmoGive 28
Weapon.SlotNumber 3
Inventory.Pickupmessage "$PU_WATERWAVE"
Obituary "$OB_WATERWAVE"
Tag "$TAG_WATERWAVE"
weapon.ammotype "WaterWaveAmmo"
inventory.icon "WATRSI"
States
{
SpawnLoop:
WEA2 M -1
loop
Ready:
WWAV E 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_WATERWAVE)
WWAV E 1 A_WeaponReady
Goto Ready+1
Deselect:
WWAV E 0
goto DeselectSwap
Select:
WWAV E 0
goto SelectSwap
Fire:
WWAV E 0 A_JumpIfNoAmmo("NoAmmo")
WWAV E 0 A_PlaySoundEx("weapons/mm5/waterwavefire","Weapon")
WWAV E 0 A_TakeInventory("WaterWaveAmmo", 2, TIF_NOTAKEINFINITE)
WWAV FFFGGG 2 A_FireCustomMissile("WaterWaveSpawner",0,0,8,0)
WWAV E 8
WWAV E 0 A_Refire
Goto Ready+1
NoAmmo:
WWAV E 0 ACS_NamedExecuteAlways("core_noammo",0)
Goto Ready+1
}
}
