actor GravityHoldWep_NormalBar : NormalBar { args 4, 239 }

actor GravityHoldWep : BaseMM8BDMWep 10079
{
//$Category MM8BDM-Weapons/AoE
//$Title Gravity Hold
//$Sprite WEA3F0
Weapon.AmmoUse 7
Weapon.AmmoGive 28
Weapon.SlotNumber 6
Inventory.Pickupmessage "$PU_GRAVITYHOLD"
Obituary "$OB_GRAVITYHOLD"
Tag "$TAG_GRAVITYHOLD"
weapon.ammotype "GravityHoldAmmo"
inventory.icon "GRAVSI"
States
{
SpawnLoop:
WEA3 F -1
loop
Ready:
FSTO A 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_GRAVITYHOLD)
FSTO A 1 A_WeaponReady
Goto Ready+1
Deselect:
FSTO A 0
goto DeselectSwap
Select:
FSTO A 0
goto SelectSwap
Fire:
FSTO A 0 A_JumpIfNoAmmo("NoAmmo")
FSTO A 0 A_FireCustomMissile("GravityHold",0,1,8,0)
FSTO BC 1
FSTO A 0 A_PlaySoundEx("weapons/mm5/gravityholdfire","Weapon")
FSTO D 45
FSTO C 2
FSTO A 0 A_Refire
Goto Ready+1
NoAmmo:
FSTO A 1 ACS_NamedExecuteAlways("core_noammo",0)
Goto Ready+1
Death:
WEA3 F 0
stop
}
}

}
}

actor GravityHoldFX : BasicGraphicEffect
{
	reactiontime 60
	var int user_angle;
	States
	{
		SpawnFrame:
		TNT1 A 0
		TNT1 A 0 A_SetUserVar("user_angle", random(0, 6))
		SpawnLoop:
		TNT1 A 0 A_SpawnItemEx("GravityHoldParticle", 1000, 0, 512, 0, 0, -60, user_angle, SXF_ABSOLUTEANGLE|SXF_NOCHECKPOSITION|SXF_WEPFXCOLOR)
		TNT1 A 0 A_SetUserVar("user_angle", user_angle + 6)
		TNT1 A 0 A_Countdown
		loop
		Death:
		TNT1 A 0
		stop
	}
}
