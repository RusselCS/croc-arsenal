actor TopSpinWep_NormalBar : NormalBar { args 52, 93 }

actor TopSpinWep : BaseMM8BDMWep 10039
{
//$Category MM8BDM-Weapons/Close
//$Title Top Spin
//$Sprite WEAPJ0
Weapon.AmmoUse 4
Weapon.AmmoGive 28
Weapon.SlotNumber 4
Inventory.Pickupmessage "$PU_TOPSPIN"
Tag "$TAG_TOPSPIN"
Obituary "$OB_TOPSPIN"
weapon.ammotype "TopSpinAmmo"
inventory.icon "TOPSI"
States
{
SpawnLoop:
WEAP J -1
loop
Ready:
TOPH A 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_TOPSPIN)
TOPH A 0 A_TakeInventory("SpinTime",1)
ReadyReal:
TOPH A 1 A_WeaponReady
Goto Ready+2
Deselect:
TOPH A 0 A_GunFlash("NoFlash")
goto DeselectSwap
Select:
TOPH A 0
goto SelectSwap
Fire:
TOPH A 0 A_JumpIfInventory("SpinTime",1,"SpinEnd")
TOPH A 0 A_JumpIfInventory("IsBot", 1, "BotCheck")
TOPH A 0 A_JumpIfNoAmmo("NoAmmo")
TOPH A 0 A_JumpIf(!CallACS("core_CheckFooting_ThruPlayer"),"SpinBegin")
goto ReadyReal
SpinBegin:
TNT1 A 0 A_GiveInventory("TakeTopSpinAmmo",1) //sync >_>
TOPH A 0 A_TakeInventory("JumpCancler", 1)
TOPH A 0 A_GunFlash //A_PlaySoundEx("weapons/mm3/topspinfire","Weapon")
SpinLoop:
TOPH A 0 A_GiveInventory("SpinTime",1)
TOPH A 0 A_SpawnItemEx("TopRadius",8,0,0,0,0,0,0,SXF_NOCHECKPOSITION)
TOPH A 0 A_JumpIfInventory("CutterFlag",999,"Ready")
//TNT1 A 1 A_JumpIf(CallACS("core_CheckFooting"),"SpinEnd")
TNT1 A 1 A_JumpIf(CallACS("core_CheckFooting_ThruPlayer"),"SpinEnd")
TOPH A 0 A_JumpIfInventory("EndSpin",1,"SpinEnd")
loop
SpinEnd:
TOPH A 0 A_GunFlash("NoFlash")
TNT1 A 0 A_TakeInventory("TopSpinFlag", 5)
TOPH A 0 A_TakeInventory("SpinTime",1)
TOPH A 0 A_TakeInventory("EndSpin",1)
TOPH A 0 A_StopSound(CHAN_WEAPON)
TOPH A 1 //ACS_NamedExecuteAlways("core_noammo",0)
TOPH A 0 A_TakeInventory("CutterFlag", 15)
//TOPH A 0 A_Refire
Goto Ready+1
BotCheck:
TOPH A 0 A_JumpIf(momz==0, 1)
Goto Fire+3
TOPH A 0 ThrustThingZ(0,50,0,1)
Goto Fire+3
NoAmmo:
TOPH A 1 ACS_NamedExecuteAlways("core_noammo",0)
Goto Ready+1
Flash:
TNT1 A 0 A_PlaySoundEx("weapons/mm3/topspinfire","Weapon")
TOPH BBCCDDEE 1 A_SetAngle(angle-20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle-20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle-20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle-20.0)
TNT1 A 0 A_GiveInventory("EndSpin",1)
NoFlash:
TNT1 A 0
stop

AltFlash:
TNT1 A 0 A_PlaySoundEx("weapons/mm3/topspinfire","Weapon")
TOPH FFGGHHII 1 A_SetAngle(angle+20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle+20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle+20.0)
TNT1 AAAAAAAAAAA 1 A_SetAngle(angle+20.0)
TNT1 A 0 A_GiveInventory("EndSpin",1)
goto NoFlash
}
}
