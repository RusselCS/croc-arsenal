actor NeedleCannonWep_NormalBar : NormalBar { args 4, 220 }

actor NeedleCannonWep : BaseMM8BDMWep 10040
{
//$Category MM8BDM-Weapons/Rapid
//$Title Needle Cannon
//$Sprite WEAPL0
Weapon.AmmoUse 1
Weapon.AmmoGive 28
Weapon.SlotNumber 3
Inventory.Pickupmessage "$PU_NEEDLECANNON"
Obituary "$OB_NEEDLECANNON"
Tag "$TAG_NEEDLECANNON"
weapon.ammotype "NeedleCannonAmmo"
inventory.icon "NEEDSI"
States
{
SpawnLoop:
WEAP L -1
loop
Ready:
NEDC B 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_NEEDLECANNON)
NEDC B 1 A_WeaponReady
Goto Ready+1
Deselect:
NEDC B 0
goto DeselectSwap
Select:
NEDC B 0
goto SelectSwap
Fire:
Hold:
NEDC B 0 A_JumpIfNoAmmo("NoAmmo")
NEDC B 0 A_PlaySoundEx("weapons/mm3/needlecannonfire","Weapon")
NEDC B 0 A_FireCustomMissile("Needle",random(-2,2),0,8,0)
NEDC CDB 2
NEDC B 0 A_PlaySoundEx("weapons/mm3/needlecannonfire","Weapon")
NEDC D 0 A_FireCustomMissile("Needle",random(-2,2),0,8,0)
NEDC CDB 2
NEDC B 0 A_PlaySoundEx("weapons/mm3/needlecannonfire","Weapon")
NEDC B 0 A_FireCustomMissile("Needle",random(-2,2),1,8,0)
NEDC CDB 2
NEDC B 0 A_Refire
Goto Ready+1
NoAmmo:
NEDC B 1 ACS_NamedExecuteAlways("core_noammo",0)
Goto Ready+1
}
}

actor NeedleCannonAmmo : Ammo
{
inventory.amount 1
inventory.maxamount 28
+INVENTORY.IGNORESKILL
}


actor Needle : BasicProjectile
{
Radius 6

damagetype "NeedleCannon"
Obituary "$OB_NEEDLECANNON"
damage (16)
speed 38
States
{
Spawn:
NEDC A 1
loop
}
}
