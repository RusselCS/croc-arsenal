actor TimeSlowWep : BaseMM8BDMWep 11010
{
//$Category MM8BDM-Weapons/AoE
//$Title Time Slow
//$Sprite SLOTX0
Weapon.AmmoUse 14
Weapon.AmmoGive 28
Weapon.SlotNumber 6
Inventory.Pickupmessage "$PU_TIMESLOW"
Obituary "$OB_TIMESLOW"
Tag "$TAG_TIMESLOW"
weapon.ammotype "TimeSlowAmmo"
Inventory.Icon "TIMMI2"
States
{
SpawnLoop:
SLOT X -1
loop
Ready:
TSLO A 0 ACS_NamedExecuteWithResult("core_weaponcolor", CLR_TIMESLOW)
TSLO A 1 A_WeaponReady
goto Ready+1
Deselect:
TSLO A 0
goto DeselectSwap
Select:
TSLO A 0
goto SelectSwap
Fire:
TSLO A 0 
TSLO A 0 A_JumpIfInventory("PowerTime",1,"Ready")
goto Fire1
Fire1:
TSLO A 0 A_JumpIfInventory("TimeSlowAmmo", 14, "Fire2")
Goto NoAmmo
Fire2:
//With Hand
TSLO A 0 A_JumpIfNoAmmo("NoAmmo")
TSLO A 0 A_JumpIfInventory("TimeSlowCooldown", 1, "NoAmmo")
TSLO A 0 A_ScaleVelocity(0.5)
SLOT JKLMN 2
TNT1 A 0 A_PlaySoundEx("weapons/mm1/timeslowfire","Weapon")
TNT1 A 0 ACS_NamedExecuteAlways("core_whiteflash",0)
TNT1 A 0 A_FireCustomMissile("TimeSlow",0,1,8,0)
TNT1 A 0 A_GiveInventory("TimeSlowCooldown", 1)
TNT1 A 0 A_SpawnItemEx("TimeSlowCooldowner")
//SLOT LMN 3
TNT1 A 2
TSLO BC 1
TSLO A 1
goto Ready+1
NoAmmo: 
TSLO A 1 ACS_NamedExecuteAlways("core_noammo",0)
goto Ready+1
NotReady:
TSLO A 1
Goto Ready+1
}
}
